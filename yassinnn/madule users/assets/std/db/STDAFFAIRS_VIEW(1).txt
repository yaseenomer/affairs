
CREATE  VIEW STDAFFAIRS_VIEW AS 
   SELECT ST.STUDENTNO,ST.STUDENTNAME,
   ST.NATIONNO,NA.NATIONNAME,ST.SEXNO,SE.SEXNAME,ST.BLOODGROUPNO,B.BLOODGROUPNAME,
ST.PROG_NO,PR.PROG_NAME,
   ST.ADMISSIONNO,A.ADMISSIONNAME,ST.STATENO,S.STATENAME,ST.PROVINCENO,PR.PROVINCENAME,
ST.YEARNO,F.YEARNAME,
   ST.NATIONID,ST.NATIONDATE,ST.ORIGINALHOME,ST.ADDRESSREM,ST.CRELATIVENAME,ST.PAYJOBNO,
PJ.PAYJOBNAME,ST.CRELATIVEADDRESS,
   ST.CRELATIVEAPHONE, ST.BIRTHDATE, ST.Nat_id,ST.TELPHONES,ST.email,ST.COLNO,C.COLNAME,
P.PAIDFEES,P.RECEIPTNO,
   P.RECEIPTDATE,P.SEMNO,SE.SEMNAME,R.TUTIONFEES,R.LOCATIONNO,L.LOCATIONNAME,R.CLASSNO,
CL.CLASSNAME,R.SEM_FINAL_RESULT
   from
   STUDENTINFORMATION ST,REGISTORCHECK R,CLASSES CL,LOCATIONS L,PAYMENTS P,SEMESTERS SI,
COLLEGES C,
   PAYJOB PJ,FEESANDYEAR F,PROVINCES PR,STATES S,ADMISSION A,PROGRAM PR,BLOODGROUP B,SEX SE,
NATIONALATIES NA
   WHERE 
     NA.NATIONNO=ST.NATIONNO AND 
     P.SEMNO = SI.SEMNO AND
     P.SEMNO = R.SEMNO AND
     ST.BLOODGROUPNO = B.BLOODGROUPNO AND
     ST.PROG_NO = PR.PROG_NO AND 
     ST.PROG_NO = R.PROG_NO AND 
     ST.ADMISSIONNO = A.ADMISSIONNO AND
     ST.ADMISSIONNO = R.ADMISSIONNO AND    
     ST.STATENO  = S.STATENO  AND
     ST.PROVINCENO = PR.PROVINCENO AND 
     ST.YEARNO = F.YEARNO AND 
     ST.YEARNO = R.YEARNO AND 
     ST.YEARNO = P.YEARNO AND 
     ST.PAYJOBNO = PJ.PAYJOBNO AND
     ST.COLNO = C.COLNO AND 
     R.LOCATIONNO = L.LOCATIONNO AND
     R.CLASSNO = CL.CLASSNO  AND
     ST.STUDENTNO=R.STUDENTNO AND
     ST.STUDENTNO=P.STUDENTNO

   F.YEARNAME,ST.NATIONID,ST.NATIONDATE,ST.ORIGINALHOME,ST.ADDRESSREM,ST.CRELATIVENAME,PJ.PAYJOBNO,PJ.PAYJOBNAME,

   P.RECEIPTNO,P.RECEIPTDATE,R.TUTIONFEES,L.LOCATIONNO,L.LOCATIONNAME,CL.CLASSNO,CL.CLASSNAME,R.SEM_FINAL_RESULT
from
   STUDENTINFORMATION ST,REGISTORCHECK R,CLASSES CL,LOCATIONS L,PAYMENTS P,SEMESTERS SI,COLLEGES C,
   PAYJOB PJ,FEESANDYEAR F,PROVINCES PR,STATES S,ADMISSION A,PROGRAM PR,BLOODGROUP B,SEX SE,NATIONALATIES NA
WHERE 
     NA.NATIONNO=ST.NATIONNO AND

     ST.PROG_NO = R.PROG_NO AND 
     ST.ADMISSIONNO = A.ADMISSIONNO AND
     ST.ADMISSIONNO = R.ADMISSIONNO AND

     ST.PROVINCENO = PR.PROVINCENO AND 
     ST.YEARNO = F.YEARNO AND 
     ST.YEARNO = R.YEARNO AND 
     ST.YEARNO = P.YEARNO AND 
     ST.PAYJOBNO = PJ.PAYJOBNO AND
     ST.COLNO = C.COLNO AND 
     R.LOCATIONNO = L.LOCATIONNO AND
     R.CLASSNO = CL.CLASSNO  AND
     ST.STUDENTNO=R.STUDENTNO AND
